(setq x 10)
(cond
    ((= x 0) 5)
    ((> x 0) 3))


(
    defun factorial (num)
        (cond 
        ((= num 0) 1)
        ((> num 0) (* num (factorial (- num 1))))
        )
)
(
    defun hola (num)
        (+ 1 num)
        (- 1 num)
        (* 1 num)
        (/ 1 num)
)
(factorial 5)
(atom 4)
(hola 5)
